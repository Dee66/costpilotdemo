# CostPilot Terminal Commands Demo
# Typical AWS Terraform workflow with CostPilot integration

# 1. Initialize Terraform
terraform init

# 2. Plan infrastructure changes
terraform plan -out=tfplan

# 3. Run CostPilot analysis on the plan
costpilot scan --plan tfplan.json --output results.json

# 4. Check for cost regressions before applying
costpilot diff --baseline baseline.json --current tfplan.json

# 5. Get detailed cost breakdown
costpilot explain --resource aws_instance.web_server --region us-east-1

# 6. Predict monthly costs
costpilot predict --plan tfplan.json --months 12

# 7. Apply with confidence (after CostPilot approval)
terraform apply tfplan

# 8. Monitor ongoing costs
costpilot trend --since 2024-01-01 --service EC2

# 9. Generate cost optimization suggestions
costpilot autofix --plan tfplan.json --safe-mode

# 10. Audit cost changes
costpilot audit --pr 42 --output audit_report.md
